#include<cstdio>
#include<iostream>
#include<string>
#include<fstream>
#include<cstring>

using namespace std;

int main(){
	string str[200][3];
	char city[200] = {};
	int cot = -1;
	char fileNameIn[30] = {};
	char fileNameOut[30] = {};
	scanf("%s",fileNameIn);
	scanf("%s",fileNameOut);
	ifstream yq(fileNameIn);
	ofstream yqout(fileNameOut, ios::trunc);
	if (!yq.is_open()){
		cout << "can not open this file" << endl;
		return 0;
	}
	for (int i = 0; i < 200; i++){
		for (int j = 0; j < 3; j++){
			yq >> str[i][j];
		}
	}
	scanf("%s",city);
	for(int i = 0; i < 200; i++){
		if(str[i][0] == city){
			cot = i;
		}
	}
	if(cot != -1){
		yqout << str[cot][0] << endl;
		for(int i = 0; i < 200; i++){
			if(str[i][0] == str[cot][0])
				yqout  << str[i][1] << "        " << str[i][2] << endl;
		}	
	}else{
		for (int i = 0; i < 200; i++){
			if(str[i][0] != str[i - 1][0]){
				if(str[i][0] != str[0][0])
					yqout << "\n" << str[i][0] << endl;
				else yqout  << str[i][0] << endl;
			}
			yqout  << str[i][1] << "        " << str[i][2] << endl;	
		}
	}
	yq.close();
	yqout.close();
	return 0;
}
